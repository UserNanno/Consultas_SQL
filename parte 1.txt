          '#LineaDisponibleNoBCPActual').on('keypress', function (e) {
            var k = String.fromCharCode(e.charCode);
            var v = this.value;
            var dp = v.indexOf('.');

            // reject illegal chars
            if ((k < '0' || k > '9') && k !== '.') return false;

            // reject any input that takes the length
            // two or more beyond the decimal point
            if (dp >= 0 && e.target.selectionStart > dp && e.target.selectionEnd > dp && v.length > dp + 2) {//v.length > dp + 2) {
                return false;
            }

            // don't accept >1 decimal point, or as first char
            if (k === '.' && (dp >= 0 || v.length === 0)) {
                return false;
            }
        });

        $('#CDECEM').on('keypress', function (e) {

         $.event.special.inputchange = {
                setup: function() {
                    var self = this, val;
                    $.data(this, 'timer', window.setInterval(function() {
                        val = self.value;
                        if ( $.data( self, 'cache') != val ) {
                            $.data( self, 'cache', val );
                            $( self ).trigger( 'inputchange' );

                teardown: function() {
                    window.clearInterval( $.data(this, 'timer') );
                },
                add: function() {
                    $.data(this, 'cache', this.value);
                }

        function addCommas(nStr)
        {
            nStr += '';
            x = nStr.split('.');
            x1 = x[0];
            x2 = x.length > 1 ? '.' + x[1] : '';
            var rgx = /(\d+)(\d{3})/;
            while (rgx.test(x1)) {
                x1 = x1.replace(rgx, '$1' + ',' + '$2');
            }
            return x1 + x2;
        }

        function selectElementContents(id) {
            var el = document.getElementById(id);
            var body = document.body, range, sel;
            if (document.createRange && window.getSelection) {
                range = document.createRange();
                sel = window.getSelection();
                sel.removeAllRanges();
                try {
                    range.selectNodeContents(el);
                    sel.addRange(range);
                } catch (e) {
                    range.selectNode(el);
                    sel.addRange(range);
                }
            } else if (body.createTextRange) {
                range = body.createTextRange();
                range.moveToElementText(el);
                range.select();
            }
            document.execCommand("Copy");
        }

        function selectElementContents(id) {
            var el = document.getElementById(id);
            var body = document.body, range, sel;
            if (document.createRange && window.getSelection) {
                range = document.createRange();
                sel = window.getSelection();
                sel.removeAllRanges();
                try {
                    range.selectNodeContents(el);
                    sel.addRange(range);
                } catch (e) {
                    range.selectNode(el);
                    sel.addRange(range);
                }
            } else if (body.createTextRange) {
                range = body.createTextRange();
                range.moveToElementText(el);
                range.select();
            }
            document.execCommand("Copy");
        }

        function VigilaActividad() {

            var segundosDesdeUltimaActividad = 0;

            var timeOut = '60';

            var maximaInactividad = timeOut * 60;

                setInterval(function () {
                    segundosDesdeUltimaActividad++;

                if (segundosDesdeUltimaActividad > maximaInactividad) {
                    location.href = '/Error/Inactividad';
                    //alert(timeOut);
                }
            }, 1000);

                segundosDesdeUltimaActividad = 0;
            }

                'keydown', 'keypress', 'scroll', 'mousedown', 'mousemove', 'touchstart', 'resize'
            ];

                EventosActividad.forEach(function (eventName) {
                document.addEventListener(eventName, reseteaActividad, true);
            });
        }
